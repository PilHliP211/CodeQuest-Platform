## What

`progressStore.ts` provides typed functions for reading and writing per-lesson progress and total XP to localStorage. It builds on top of `storage.ts` (the generic wrapper from E-02) ‚Äî it does not call `localStorage` directly.

Key schema:
- Per-lesson: `codequest:progress:{packId}:{lessonId}`
- Total XP: `codequest:xp`

## Tasks

1. Create `src/engine/progressStore.ts`:
   ```ts
   import { storageGet, storageSet, storageRemove } from './storage';
   import type { LessonProgress } from '@/types/progress';

   function progressKey(packId: string, lessonId: string): string {
     return `codequest:progress:${packId}:${lessonId}`;
   }

   const XP_KEY = 'codequest:xp';

   export function getLessonProgress(
     packId: string,
     lessonId: string,
   ): LessonProgress | null {
     const raw = storageGet(progressKey(packId, lessonId));
     if (raw === null) return null;
     // Basic shape validation ‚Äî if malformed, return null
     if (
       typeof raw !== 'object' ||
       typeof (raw as Record<string, unknown>).phase1Complete !== 'boolean'
     ) {
       return null;
     }
     return raw as LessonProgress;
   }

   export function saveLessonProgress(
     packId: string,
     lessonId: string,
     progress: LessonProgress,
   ): void {
     storageSet(progressKey(packId, lessonId), progress);
   }

   export function getTotalXP(): number {
     const raw = storageGet(XP_KEY);
     if (typeof raw === 'number') return raw;
     return 0;
   }

   export function addXP(amount: number): void {
     const current = getTotalXP();
     storageSet(XP_KEY, current + amount);
   }

   export function resetAllProgress(): void {
     // Remove all progress keys ‚Äî iterate localStorage keys matching the prefix
     const keysToRemove: string[] = [];
     for (let i = 0; i < localStorage.length; i++) {
       const key = localStorage.key(i);
       if (key !== null && key.startsWith('codequest:progress:')) {
         keysToRemove.push(key);
       }
     }
     for (const key of keysToRemove) {
       storageRemove(key);
     }
     storageRemove(XP_KEY);
   }
   ```

2. `storageGet`, `storageSet`, `storageRemove` are the functions from `storage.ts` (E-02, S-02.02). If `storageRemove` doesn't exist yet, add it to `storage.ts` in this story.

3. `resetAllProgress` does NOT remove the profile key (`codequest:profile`). The learner's name is preserved.
