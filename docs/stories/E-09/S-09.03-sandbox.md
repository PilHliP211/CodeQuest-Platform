# S-09.03: `sandbox.ts` — Whitelist Enforcement

**Epic:** E-09
**Size:** M
**Dependencies:** S-09.02 (AST types)

## Skills to Load

- `security-rules` — no eval, no DOM access, whitelist enforcement
- `typescript-standards` — typed error, no `any`
- `error-handling` — custom error class, not plain strings

## What

`sandbox.ts` performs a full AST traversal before any execution to reject code that accesses disallowed identifiers or uses forbidden constructs. This is the security gate — it runs before the interpreter so that no dangerous code ever reaches execution.

The sandbox validates the *whole program* in one pass. Any violation throws a `SandboxError` with a descriptive message and line number. The interpreter calls `validateAST` at the start of every execution.

## Tasks

1. Create `src/engine/interpreter/sandbox.ts`:
   ```ts
   import type { ASTNode } from './astTypes';

   export class SandboxError extends Error {
     constructor(
       message: string,
       public readonly line?: number,
     ) {
       super(message);
       this.name = 'SandboxError';
     }
   }

   const FORBIDDEN_GLOBALS = new Set([
     'window', 'document', 'localStorage', 'sessionStorage',
     'fetch', 'XMLHttpRequest', 'WebSocket', 'eval', 'Function',
     '__proto__', 'constructor', 'prototype', 'globalThis', 'self',
     'navigator', 'location', 'history', 'setTimeout', 'setInterval',
     'clearTimeout', 'clearInterval', 'alert', 'confirm', 'prompt',
     'require', 'module', 'exports', 'process',
   ]);

   /**
    * Walks the AST and throws SandboxError if any forbidden construct is found.
    * @param ast - The parsed program node from acorn
    * @param allowedIdentifiers - The lesson's availableFunctions names
    */
   export function validateAST(ast: ASTNode, allowedIdentifiers: string[]): void {
     walkNode(ast, new Set(allowedIdentifiers), new Set<string>());
   }

   function walkNode(
     node: ASTNode,
     allowedAPIs: Set<string>,
     localVars: Set<string>,
   ): void {
     if (node === null || typeof node !== 'object') return;

     switch (node.type) {
       case 'Program':
         for (const stmt of (node as { body: ASTNode[] }).body) {
           walkNode(stmt, allowedAPIs, new Set(localVars));
         }
         break;

       case 'VariableDeclaration': {
         if (node.kind !== 'let') {
           throw new SandboxError(
             `Use 'let' to declare variables, not '${node.kind}'.`,
             (node as { loc?: { start: { line: number } } }).loc?.start.line,
           );
         }
         for (const decl of node.declarations) {
           if (decl.id.type === 'Identifier') localVars.add(decl.id.name);
           if (decl.init !== null) walkNode(decl.init, allowedAPIs, localVars);
         }
         break;
       }

       case 'Identifier': {
         const name = node.name;
         if (
           !allowedAPIs.has(name) &&
           !localVars.has(name) &&
           FORBIDDEN_GLOBALS.has(name)
         ) {
           throw new SandboxError(
             `'${name}' is not available in CodeQuest.`,
             (node as { loc?: { start: { line: number } } }).loc?.start.line,
           );
         }
         break;
       }

       case 'MemberExpression': {
         // Reject any property access on forbidden objects
         const obj = (node as { object: ASTNode; property: ASTNode }).object;
         if (obj.type === 'Identifier' && FORBIDDEN_GLOBALS.has(obj.name)) {
           throw new SandboxError(
             `Access to '${obj.name}' is not allowed.`,
             (node as { loc?: { start: { line: number } } }).loc?.start.line,
           );
         }
         break;
       }

       case 'ImportDeclaration':
       case 'ExportNamedDeclaration':
       case 'ExportDefaultDeclaration':
         throw new SandboxError('import/export is not supported in CodeQuest.');

       default: {
         // Walk all child nodes generically
         for (const value of Object.values(node)) {
           if (Array.isArray(value)) {
             for (const item of value) {
               if (item !== null && typeof item === 'object' && 'type' in item) {
                 walkNode(item as ASTNode, allowedAPIs, localVars);
               }
             }
           } else if (value !== null && typeof value === 'object' && 'type' in value) {
             walkNode(value as ASTNode, allowedAPIs, localVars);
           }
         }
       }
     }
   }
   ```

2. Note: `localVars` is threaded through the walker to allow user-declared `let` variables as valid identifiers. Each `BlockStatement` and `FunctionDeclaration` creates a new child scope via `new Set(localVars)`.

3. Parse options must include `{ locations: true }` in `acorn.parse()` to populate `loc` for line numbers in errors.
