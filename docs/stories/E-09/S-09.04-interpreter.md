# S-09.04: `interpreter.ts` — AST Walker Core

**Epic:** E-09
**Size:** M
**Dependencies:** S-09.03 (`sandbox.ts`)

## Skills to Load

- `security-rules` — no eval; scope is an injected object only
- `typescript-standards` — typed scope, `unknown` not `any`, discriminated union returns
- `error-handling` — runtime errors return `ExecutionResult`, never throw to the caller

## What

`interpreter.ts` is the execution engine. It parses player code with `acorn`, runs `validateAST` from the sandbox, then walks the AST and evaluates each node. The execution scope is a plain `Record<string, unknown>` injected by the caller — the interpreter cannot access any platform internals.

The interpreter handles the simplified-JS subset defined in CONTENT_SCHEMA.md: `let` declarations, function calls, `for` loops, `if`/`else`, arithmetic, and function definitions. It does not handle classes, `async`/`await`, generators, or any ES module syntax.

## Tasks

1. Create `src/engine/interpreter/interpreter.ts`:

   ```ts
   import * as acorn from 'acorn';
   import { validateAST, SandboxError } from './sandbox';
   import type { ASTNode } from './astTypes';

   export interface PlayerFriendlyError {
     message: string;
     line?: number;
   }

   export type ExecutionResult =
     | { success: true }
     | { success: false; error: PlayerFriendlyError };

   const MAX_LOOP_ITERATIONS = 10_000;

   export function execute(
     code: string,
     scope: Record<string, unknown>,
   ): ExecutionResult {
     // 1. Parse
     let ast: acorn.Node;
     try {
       ast = acorn.parse(code, { ecmaVersion: 2020, locations: true });
     } catch (err) {
       return { success: false, error: parseFriendlyError(err) };
     }

     // 2. Sandbox check
     try {
       validateAST(ast as unknown as ASTNode, Object.keys(scope));
     } catch (err) {
       if (err instanceof SandboxError) {
         return { success: false, error: { message: err.message, line: err.line } };
       }
       throw err;
     }

     // 3. Execute
     try {
       const env = { ...scope };
       executeBlock((ast as { body: ASTNode[] }).body, env);
       return { success: true };
     } catch (err) {
       if (err instanceof RuntimeError) {
         return { success: false, error: { message: err.message, line: err.line } };
       }
       throw err;
     }
   }

   class RuntimeError extends Error {
     constructor(message: string, public readonly line?: number) {
       super(message);
       this.name = 'RuntimeError';
     }
   }

   function executeBlock(body: ASTNode[], env: Record<string, unknown>): unknown {
     for (const node of body) {
       const result = executeNode(node, env);
       if (result instanceof ReturnSignal) return result.value;
     }
     return undefined;
   }

   class ReturnSignal { constructor(public readonly value: unknown) {} }

   function executeNode(node: ASTNode, env: Record<string, unknown>): unknown {
     switch (node.type) {
       case 'ExpressionStatement':
         return executeNode(node.expression, env);

       case 'CallExpression': {
         const callee = executeNode(node.callee, env);
         if (typeof callee !== 'function') {
           const name = node.callee.type === 'Identifier' ? node.callee.name : '(unknown)';
           throw new RuntimeError(
             `'${name}' is not a function available here. Check your available functions.`,
             node.loc?.start.line,
           );
         }
         const args = node.arguments.map((arg) => executeNode(arg, env));
         return (callee as (...a: unknown[]) => unknown)(...args);
       }

       case 'Identifier':
         if (node.name in env) return env[node.name];
         throw new RuntimeError(`'${node.name}' is not defined.`, node.loc?.start.line);

       case 'Literal':
         return node.value;

       case 'BinaryExpression':
         return evalBinary(node.operator, executeNode(node.left, env), executeNode(node.right, env));

       case 'VariableDeclaration':
         for (const decl of node.declarations) {
           const val = decl.init !== null ? executeNode(decl.init, env) : undefined;
           if (decl.id.type === 'Identifier') env[decl.id.name] = val;
         }
         return undefined;

       case 'AssignmentExpression': {
         const val = executeNode(node.right, env);
         if (node.left.type === 'Identifier') env[node.left.name] = val;
         return val;
       }

       case 'BlockStatement':
         return executeBlock(node.body, { ...env });

       case 'IfStatement': {
         const cond = executeNode(node.test, env);
         if (cond) return executeNode(node.consequent, env);
         if (node.alternate !== null) return executeNode(node.alternate, env);
         return undefined;
       }

       case 'ForStatement': {
         const loopEnv = { ...env };
         if (node.init !== null) executeNode(node.init, loopEnv);
         let iterations = 0;
         while (node.test === null || executeNode(node.test, loopEnv)) {
           if (++iterations > MAX_LOOP_ITERATIONS) {
             throw new RuntimeError('Loop ran too many times. Check your loop condition.');
           }
           executeNode(node.body, loopEnv);
           if (node.update !== null) executeNode(node.update, loopEnv);
         }
         return undefined;
       }

       case 'FunctionDeclaration': {
         const fn = (...args: unknown[]) => {
           const fnEnv = { ...env };
           node.params.forEach((param, i) => {
             if (param.type === 'Identifier') fnEnv[param.name] = args[i];
           });
           const result = executeBlock(node.body.body, fnEnv);
           return result instanceof ReturnSignal ? result.value : undefined;
         };
         env[node.id.name] = fn;
         return undefined;
       }

       case 'ReturnStatement':
         return new ReturnSignal(node.argument !== null ? executeNode(node.argument, env) : undefined);

       default:
         throw new RuntimeError(`Unsupported syntax: ${node.type}`);
     }
   }

   function evalBinary(op: string, left: unknown, right: unknown): unknown {
     switch (op) {
       case '+': return (left as number) + (right as number);
       case '-': return (left as number) - (right as number);
       case '*': return (left as number) * (right as number);
       case '/': return (left as number) / (right as number);
       case '%': return (left as number) % (right as number);
       case '<': return (left as number) < (right as number);
       case '>': return (left as number) > (right as number);
       case '<=': return (left as number) <= (right as number);
       case '>=': return (left as number) >= (right as number);
       case '===': return left === right;
       case '!==': return left !== right;
       default: throw new RuntimeError(`Unsupported operator: ${op}`);
     }
   }

   function parseFriendlyError(err: unknown): PlayerFriendlyError {
     if (err !== null && typeof err === 'object' && 'message' in err) {
       const msg = String((err as { message: unknown }).message);
       const lineMatch = /\((\d+):\d+\)/.exec(msg);
       const line = lineMatch !== undefined ? parseInt(lineMatch[1], 10) : undefined;
       return { message: `Syntax error on line ${line ?? '?'}: check for missing brackets or typos.`, line };
     }
     return { message: 'Something went wrong. Check your code for typos.' };
   }
   ```

2. The `env` is always a shallow copy (`{ ...scope }`) — the caller's scope object is never mutated.

3. `MAX_LOOP_ITERATIONS = 10_000` prevents infinite loops from freezing the browser. Players writing `for (let i = 0; i < 3; i++)` will never hit this; it's only triggered by bugs like `for (let i = 0; i < 10000; i++)` or `while (true)`.
